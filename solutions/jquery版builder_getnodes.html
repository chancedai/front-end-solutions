        var getNodes = function(wrap,attr) {
            attr = attr||'node-type';
            wrap = $(wrap);
            var nodes = $("[" + attr + "]", wrap);
            var nodesObj = {};
            nodesObj.wrap = wrap;
            nodes.each(function(i) {
                var item = $(this);
                nodesObj[item.attr(attr)] = item;
            });
            return nodesObj;
        };
